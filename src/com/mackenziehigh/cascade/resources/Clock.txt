
root = [ task * ];

task = after / every;

after = [ "after" , delay , units , "send" , value , "to" , queue , "every" , period , units ];

every = [ "send" , value , "to" , queue , "every" , period , units ];

value = $ANY;

delay = $INT;

period = $INT;

units = "days"
      / "hours"
      / "minutes"
      / "seconds"
      / "milliseconds"
      ;

queue = $ATOM;
